<task type="auto">
  <name>Task 2: Add remote and push `main` branch</name>
  <files>.git/config</files>
  <read_first>
    - .git/config
  </read_first>
  <action>
    Add a remote named `origin` pointing to `git@github.com:stevmachine/skills.git`. If a remote named `origin` already exists with a different URL, update it. Push the local `main` branch to `origin/main` with `git push -u origin main`.

  <acceptance_criteria>
    - `.git/config` contains `origin` with `git@github.com:stevmachine/skills.git`
    - `git branch -r` shows `origin/main`
    - `git rev-parse origin/main` matches the local `main` HEAD
  </acceptance_criteria>
  <verify>
    <automated>git remote get-url origin | grep -q 'git@github.com:stevmachine/skills.git' && git fetch origin && git branch -r | grep 'origin/main'</automated>
  </verify>
  <done>Local `main` branch is pushed to the GitHub repository.</done>
</task>
